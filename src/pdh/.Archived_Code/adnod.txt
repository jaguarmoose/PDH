def ns2path(ns):    # convert a nodestring to dir path

    nanc= ns.split(":")
    nlev = len(nanc)
    for I in range(1,nlev):

def fib2(n):   # return Fibonacci series up to n
    result = []
    a, b = 0, 1
    while b < n:
        result.append(b)
        a, b = b, a+b
    return result
